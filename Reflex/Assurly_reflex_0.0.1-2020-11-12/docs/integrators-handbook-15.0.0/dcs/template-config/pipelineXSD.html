Document creation pipeline XML structure format file

<?xml version="1.0" encoding="UTF-8"?>
<xsd:schema xmlns="http://www.hannover-re.com/dcs/schema/pipeline"
    xmlns:xsd="http://www.w3.org/2001/XMLSchema"
    xmlns:beans="http://www.springframework.org/schema/beans"
    targetNamespace="http://www.hannover-re.com/dcs/schema/pipeline"
    elementFormDefault="qualified"
    attributeFormDefault="unqualified">

  <xsd:import namespace="http://www.springframework.org/schema/beans"/>

  <xsd:element name="pipeline">
     <xsd:annotation>
      <xsd:documentation>
        Generic document creation pipeline. Document creation is done by sequentially executing the child processors.<br/>
        <ul>
           <li>Input  Type: Input Type of first child processor</li>
           <li>Output Type: Output Type of last child processor</li>
        </ul>
      </xsd:documentation>
    </xsd:annotation>
     <xsd:complexType>
       <xsd:choice minOccurs="1" maxOccurs="unbounded">
         <xsd:element ref="jsonToXml"/>
         <xsd:element ref="xslTransformation"/>
         <xsd:element ref="i18NXmlProcessor"/>
         <xsd:element ref="fopProcessor"/>
       </xsd:choice>
       <xsd:attribute name="id" use="required" type="xsd:ID">
         <xsd:annotation>
           <xsd:documentation>Please provide the template (folder) name as id.</xsd:documentation>
         </xsd:annotation>
       </xsd:attribute>
     </xsd:complexType>
  </xsd:element>


  <!-- The abstract processor type to hold common attributes. -->
  <xsd:complexType name ="processorType">
    <xsd:complexContent>
      <xsd:extension base="beans:identifiedType">
        <xsd:attribute name="name" use="optional" type="xsd:string">
          <xsd:annotation>
            <xsd:documentation>Optional name attribute is used to identify this
              processor element. This name will be used for the 'exitAfter' feature
              used for testing. The name must be unique within one pipeline.
            </xsd:documentation>
          </xsd:annotation>
        </xsd:attribute>
      </xsd:extension>
    </xsd:complexContent>
  </xsd:complexType>

  <xsd:element name="jsonToXml">
    <xsd:complexType>
      <xsd:annotation>
        <xsd:documentation>
        Generic json to xml conversion processor.<br/>
          <ul>
             <li>Input  Type: JSON</li>
             <li>Output Type: XML</li>
          </ul>
        </xsd:documentation>
      </xsd:annotation>
      <xsd:complexContent>
        <xsd:extension base="processorType">
        </xsd:extension>
      </xsd:complexContent>
    </xsd:complexType>
  </xsd:element>

  <xsd:element name="xslTransformation">
    <xsd:annotation>
       <xsd:documentation>
         Generic XSLT transformation processor that applys the configured 'styleSheet'. The style sheet will be loaded
         from the template directory. If the style sheet does not exist it will be tried to be loaded from the common
         directory located in the config root directory.<br/>
         <ul>
            <li>Input  Type: XML</li>
            <li>Output Type: XML</li>
         </ul>
       </xsd:documentation>
    </xsd:annotation>
    <xsd:complexType>
      <xsd:complexContent>
        <xsd:extension base="processorType">
          <xsd:attribute name="styleSheet" use="required" type="xsd:string">
            <xsd:annotation>
              <xsd:documentation>The (XSLT 1.0) style sheet to be used by this processor.</xsd:documentation>
            </xsd:annotation>
          </xsd:attribute>
        </xsd:extension>
      </xsd:complexContent>
    </xsd:complexType>
  </xsd:element>

   <xsd:element name="i18NXmlProcessor">
    <xsd:complexType>
     <xsd:annotation>
       <xsd:documentation>
         The i18n xml processor loads the language file based on the 'language' query parameter from the template folder
         like "i18n_{lang}.xml". If the corresponding  language file does not exist the default translation file
         "i18n.xml" will be loaded". The output will produces XML that contains the loaded translation file and the input xml.<br/>

         <ul>
            <li>Input  Type: XML</li>
            <li>Output Type: XML</li>
         </ul>

       </xsd:documentation>
     </xsd:annotation>
      <xsd:complexContent>
        <xsd:extension base="processorType">
        </xsd:extension>
      </xsd:complexContent>
    </xsd:complexType>
  </xsd:element>

  <xsd:element name="fopProcessor">
    <xsd:complexType>
      <xsd:complexContent>
        <xsd:extension base="processorType">
          <xsd:annotation>
            <xsd:documentation>
              Fop processor that creates a defined output from the input FO XML. The output has to be defined by the "mimeType" attribute<br/>
              <ul>
                <li>Input  Type: XML</li>
                <li>Output Type: defined by "mimeType" attribute</li>
              </ul>
            </xsd:documentation>
          </xsd:annotation>
          <xsd:attribute name="mimeType" type="xsd:string" use="required">
            <xsd:annotation>
              <xsd:documentation>MimeType of the output format. Must be a value of org.apache.fop.apps.MimeConstants.<br/>
                Tested values are:<br/>
                <ul>
                  <li>application/pdf</li>
                  <li>application/rtf</li>
                  <li>text/richtext</li>
                  <li>text/rtf</li>
                </ul>
              </xsd:documentation>
            </xsd:annotation>
          </xsd:attribute>
        </xsd:extension>
      </xsd:complexContent>
    </xsd:complexType>
  </xsd:element>

</xsd:schema>
