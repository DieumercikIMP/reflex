Configuration

This is a reference configuration for the RSP that assumes that all ReFlex components (RSP, CEP, RAS, DCS) are installed on the same server instance. However, it is not necessary to run all applications on the same server instance. In case the components are installed on different instances, please make sure that, all configured URL items listed below correctly point to the required ReFlex component (CEP, RAS, DCS) and that they are accessible from the RSP installation.

### Internal or External Prepare

# Enable or disable the internal prepare mechanism. It is not possible to use both modes at the same time.
# If internal prepare is enabled, external prepare is disabled and vice versa.
com.hannoverre.reflex.rsp.integration.rest.internalPrepareAllowed=true

# If a multi assessment is allowed, on external prepare an assessment will be added to an existing user context.
# This option is only available in combination with the external prepare mode.
com.hannoverre.reflex.rsp.integration.rest.multiAssessmentsAllowed=false

# Overwrite an existing user context with a new one if external prepare is triggered.
# This option is only available in combination with the external prepare mode and the multi assessment mode.
com.hannoverre.reflex.rsp.integration.rest.allowUserContextOverwrite=false

### RSP Result Page Options ###

# If true, the ratings of the riders appear at the result page.
com.hannoverre.reflex.rsp.result.showRiderRatings=true

### RSP Automated Periodic Housekeeping ###

# Activates the scheduled housekeeping (true, false), the default is false if commented out/removed.
com.hannoverre.reflex.rsp.housekeeping.housekeepingActive=false
# The number of hours that an assessment in any state at least is stored until the housekeeping will clean up the data. Mandatory if housekeeping is active.
com.hannoverre.reflex.rsp.housekeeping.cleanUpAbandonedAssessmentsAfterHours=14
# The number of hours that a processed assessment at least is stored until the housekeeping will clean up the data. Mandatory if housekeeping is active.
com.hannoverre.reflex.rsp.housekeeping.cleanUpProcessedAssessmentsAfterHours=24
# The number of days that a user context without any assessment at least is stored until the housekeeping will clean up the data. Mandatory if housekeeping is active.
com.hannoverre.reflex.rsp.housekeeping.cleanUpUsersAfterDays=90
# Repetition interval for scheduled housekeeping in minutes. Defaults to 1440 (i.e. once every day) if commented out/removed.
com.hannoverre.reflex.rsp.housekeeping.repetitionIntervalMinutes=1440
# Starting hour for the scheduled housekeeping. Defaults to 0 (i.e. 12 a.m.) if commented out/removed.
com.hannoverre.reflex.rsp.housekeeping.startHour=3
# Starting minute for the scheduled housekeeping. Defaults to 0 if commented out/removed.
com.hannoverre.reflex.rsp.housekeeping.startMinute=30

### Authentication and JWT Configuration ###

# The fully-qualified name of the implementing class of the [authentication interface](../interfaces/authentication-interface.html).
com.hannoverre.reflex.rsp.services.authentication.implementationClassName=com.hannoverre.reflex.rsp.authentication.demo.DemoAuthentication

# The qualifier of the created token for the RSP context.
com.hannoverre.reflex.rsp.services.authentication.tokenScopes=reflex.rsp.portal

# Set the secure flag to the cookie.
com.hannoverre.reflex.rsp.services.authentication.useSecureCookie=false

# Avoiding cookies to be attached to cross-site requests. Options: Strict, Lax, None
# see https://tools.ietf.org/html/draft-ietf-httpbis-cookie-same-site-00
com.hannoverre.reflex.rsp.services.authentication.sameSiteSetting=Strict

### CEP Library Configuration ###

com.hannoverre.reflex.cep.security.keystore=file:${REFLEX_CONFIG_ROOTDIR}/cep-core/keystore.json
com.hannoverre.reflex.cep.security.jwsAlgorithmWhiteList=HS256,HS512,RS256,RS512
com.hannoverre.reflex.cep.security.allowedClockSkew=3600
com.hannoverre.reflex.cep.security.signing.kid=cep
com.hannoverre.reflex.cep.security.signing.notBeforeTimeOffset=60
com.hannoverre.reflex.cep.security.signing.algorithm=HS256
com.hannoverre.reflex.cep.security.signing.issuer=CEP
com.hannoverre.reflex.cep.security.encryption.kid=cep_external_suspend
com.hannoverre.reflex.cep.security.encryption.algorithm=A256GCM
com.hannoverre.reflex.cep.service.integration.signing.tokenScopes=reflex.cep.assessment
com.hannoverre.reflex.cep.service.integration.signing.expirationTime=3600

### Persistence Configuration ###

com.hannoverre.reflex.rsp.datasource.url=jdbc:h2:mem:rsp;DB_CLOSE_DELAY=-1
com.hannoverre.reflex.rsp.datasource.driver=org.h2.Driver
com.hannoverre.reflex.rsp.datasource.username=sa
com.hannoverre.reflex.rsp.datasource.password=

com.hannoverre.reflex.rsp.hibernate.dialect=org.hibernate.dialect.H2Dialect
com.hannoverre.reflex.rsp.hibernate.show_sql=false
com.hannoverre.reflex.rsp.hibernate.hbm2ddl.auto=update

### Authentication Configuration ###

# The key-value pair of specific locations to navigate for login and logout.
com.hannoverre.reflex.rsp.auth.implconfig={loginlocation:"http://localhost:8080/demo-auth/demo-auth.html", logoutlocation:"http://localhost:8080/rsp/auth/logout/verify"}

### Output Interfaces Configuration ###

com.hannoverre.reflex.rsp.output.implClass=com.hannoverre.reflex.rsp.output.DemoOutputServiceImpl
com.hannoverre.reflex.rsp.output.implconfig={outputDir:"${CATALINA_TMPDIR}/"}
com.hannoverre.reflex.rsp.output.miDigestPath=${CATALINA_TMPDIR}

### CEP Integration ###

# This URL template has to point to the cep installation.
com.hannoverre.reflex.rsp.integration.rest.urlTemplate=http://localhost:8080/cep-core/integration/{assessmentId}

# The CEP Integration URL
com.hannoverre.reflex.rsp.integration.rest.url=http://localhost:8080/cep-core/integration

### RAS Integration ###

# The RAS integration URL
com.hannoverre.reflex.rsp.util.ras.rest.baseUrl=http://localhost:8080/ras-web

# RAS Authorization
com.hannoverre.reflex.rsp.util.ras.user=user
com.hannoverre.reflex.rsp.util.ras.password=password

### DCS Integration ###

# The DCS Integration URL
com.hannoverre.reflex.rsp.integration.rest.document.urlTemplate=http://localhost:8080/dcs/document?type={docType}&language={lang}

# DCS Template Names
com.hannoverre.reflex.rsp.integration.rest.document.externalDocType=sample-application
com.hannoverre.reflex.rsp.integration.rest.document.internalDocType=sample-application
com.hannoverre.reflex.rsp.integration.rest.document.signDocType=sample-application
